package comp3111.popnames;

public class InputValidationForUnitTest {

	//Validating Inputs for Task 2
	public static boolean isYearValid(int Year1, int Year2) {
		if(Year1>=1880 && Year1 <= 2019 && Year2>=1880 && Year2 <= 2019 && Year2>=Year1) {
			return true;
		} else {
			return false;
		}
	}
	
	public static boolean isKValid(int K) {
		if(K>=1 && K<=1000) {
			return true;
		} else {
			return false;
		}
	}

	public static boolean isGenderValid(char Gender) {
		if(Gender == 'M' || Gender == 'F') {
			return true;
		} else {
			return false;
		}
	}
}
